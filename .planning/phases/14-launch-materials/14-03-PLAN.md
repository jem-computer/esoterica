**Option A: Use Sharp script (programmatic)**
Create `brand/scripts/generate-carousel.ts`:
```typescript
import sharp from "sharp";
import path from "path";
import { mkdirSync } from "fs";

const SLIDE_WIDTH = 1080;
const SLIDE_HEIGHT = 1350;

async function generateCarouselSlides() {
  const outputDir = path.resolve(__dirname, "../launch/carousel");
  mkdirSync(outputDir, { recursive: true });

  const heroPath = path.resolve(__dirname, "../../docs/hero-primary.png");

  // Slide 1: Hero crop with "Ancient patterns, new paths" overlay
  // Slide 2: "What if your AI agent could draw tarot cards?"
  // Slide 3: Card draw visualization
  // Slide 4: Interpretation example
  // Slide 5: "npm install -g esoterica"
  // Slide 6: "Try it today" + landing page URL

  // For each slide, create 1080x1350 canvas with brand colors
  // Use sharp composite for hero image crops
  // Text overlays require additional library - recommend Canva for final text

  for (let i = 1; i <= 6; i++) {
    await sharp({
      create: {
        width: SLIDE_WIDTH,
        height: SLIDE_HEIGHT,
        channels: 4,
        background: { r: 250, g: 249, b: 246, alpha: 1 } // #FAF9F6
      }
    })
    .png()
    .toFile(path.join(outputDir, `slide-0${i}.png`));

    console.log(`Created placeholder slide ${i}`);
  }
}

generateCarouselSlides();
```
